Write A Program to Draw the Co-Ordinate Axis At The Center Of Screen

#include<conio.h>
#include<math.h>
#include<graphics.h>
void main()
{
  int gd=DETECT,gm;
  initgraph(&gd,&gm,"C:\\TC\\BGI");
  clrscr();
  int x=0,y=0,yhalf,xhalf,ymax,xmax;
  ymax=getmaxy();
  xmax=getmaxx();
  yhalf=ymax/2;
  xhalf=xmax/2;
  line(x,yhalf,xmax,yhalf);
  line(xhalf,y,xhalf,ymax);
  getch();
  closegraph();
}

Write A Program To Create a Rainbow

#include<conio.h>
#include<math.h>
#include<graphics.h>
#include<dos.h>
void main()
{
  int gd=DETECT,gm;
  initgraph(&gd,&gm,"C:\\TC\\BGI");
  clrscr();
  setbkcolor(WHITE);
  int x=getmaxx()/2;
  int y=getmaxy()-50;
  int radius=100;
  int colors[]={RED,LIGHTRED,YELLOW,GREEN,BLUE,CYAN};
  int n=sizeof(colors)/sizeof(colors[0]);
  for(int i=0;i<n;i++)
  {
    setcolor(colors[i]);
    setfillstyle(SOLID_FILL,colors[i]);
    pieslice(x,y,0,180,radius-i*15);
    delay(100);
  }
  getch();
  closegraph();
}

Develop A Program For Themed Point Circle Drawing Algorithm On C.G

#include<conio.h>
#include<math.h>
#include<graphics.h>
#include<iostream.h>
#include<dos.h>
void main()
{
  int gd=DETECT,gm;
  initgraph(&gd,&gm,"C:\\TC\\BGI");
  clrscr();
  int xc,yc,r;
  cout<<"Enter The Center Points Of Circle : ";
  cin>>xc;
  cin>>yc;
  cout<<"Enter The Radius : ";
  cin>>r;
  int x=0;
  int y=r;
  int p=1-r;
  while(x<y)
  {
    if(p<0)
    {
      x++;
      p=p+2*x+1;
    }
    else
    {
      x++;
      y--;
      p=p+2*x+1-2*y;
    }
    putpixel(xc+x,yc-y,RED);
    putpixel(xc-x,yc-y,RED);
    putpixel(xc+x,yc+y,RED);
    putpixel(xc-x,yc+y,RED);
    putpixel(xc+y,yc+x,RED);
    putpixel(xc-y,yc+x,RED);
    putpixel(xc+y,yc-x,RED);
    putpixel(xc-y,yc-x,RED);
  }
  getch();
  closegraph();
}

Write A Program For DDA Line Algorithm

#include<conio.h>
#include<math.h>
#include<graphics.h>
#include<dos.h>
#include<iostream.h>
void main()
{
  int gd=DETECT,gm;
  initgraph(&gd,&gm,"C:\\TC\\BGI");
  clrscr();
  int x1,y1,x2,y2,dx,dy,x,y,d;
  cout<<"Enter The 1st Co-ordinates: ";
  cin>>x1>>y1;
  cout<<"Enter The 2nd Co-ordinates: ";
  cin>>x2>>y2;
  dx = x2 - x1;
  dy = y2 - y1;
  d = 2*dy - dx;
  x = x1;
  y = y1;
  putpixel(x,y,RED);
  while(x<=x2)
  {
    if(d>0)
    {
      y++;
      d = d + 2*dy - 2*dx;
    }
    else
    {
      d = d + 2*dy;
    }
    x++;
    putpixel(x,y,RED);
  }
  getch();
  closegraph();
}
