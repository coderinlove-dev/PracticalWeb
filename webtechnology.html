Explain Associative Array With Variables

<!DOCTYPE HTML>
<html>
<head>
  <title>Associative Array in Javascript</title>
</head>
<body>
  <script language="javascript">
    let name = "Arya";
    var age = "23";
    let city = "Mumbai";
    let person = {
      "personname": name,
      "personage": age,
      "personcity": city
    };
    document.write("Name : " + person.personname);
    document.write("<br>Age : " + person.personage);
    document.write("<br>City : " + person.personcity);
  </script>
</body>
</html>
